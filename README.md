##### Uipath convert string to Integer
Convert.ToInt16("12") + Convert.ToInt16("13") = 25
cint("12") + cint("13") = 25<br>
Integer.Parse("13") + Integer.Parse("12") = 25<br>

##### UiPath concatenate 2 or more strings
"A"&"B" = AB
"A"+"B" = AB<br>
string.Concat("A","B") = AB<br>
